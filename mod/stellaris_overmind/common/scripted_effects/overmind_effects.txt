# ============================================================
# Stellaris Overmind — Scripted Effects (Action Executor)
#
# Each macro action from the LLM is mapped to a scripted effect
# that uses Stellaris' built-in AI weights and effects.
#
# These are deliberately simple — the LLM decides WHAT to do,
# Stellaris' micro-AI handles HOW to do it.
# ============================================================


# --- EXPAND: Build outposts, claim systems ---
overmind_action_expand = {
	log = "[OVERMIND] Action: EXPAND"
	# Increase expansion priority via country modifier
	add_modifier = {
		modifier = overmind_expansion_focus
		days = 180
	}
}
