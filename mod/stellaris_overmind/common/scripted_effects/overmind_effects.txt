# ============================================================
# Stellaris Overmind — Scripted Effects (Action Executor)
#
# Each macro action from the LLM is mapped to a scripted effect.
# The LLM decides WHAT to do; Stellaris' micro-AI handles HOW.
#
# IMPORTANT: Each action FIRST removes all previous overmind
# modifiers to prevent stacking, then applies the new one.
# ============================================================


# --- Helper: remove all overmind modifiers before applying new ---
overmind_clear_modifiers = {
	remove_modifier = overmind_expansion_focus
	remove_modifier = overmind_military_focus
	remove_modifier = overmind_economy_focus
	remove_modifier = overmind_research_focus
	remove_modifier = overmind_diplomacy_focus
	remove_modifier = overmind_war_preparation
	remove_modifier = overmind_defense_focus
	remove_modifier = overmind_consolidation
	remove_modifier = overmind_colonize_focus
	remove_modifier = overmind_starbase_focus
	remove_modifier = overmind_espionage_focus
	# Clear personality stance flags
	remove_country_flag = overmind_aggressive
	remove_country_flag = overmind_defensive
	remove_country_flag = overmind_full_assault
	# Clear fleet sub-type flags
	remove_country_flag = overmind_fleet_corvettes
	remove_country_flag = overmind_fleet_battleships
	remove_country_flag = overmind_fleet_balanced
	# Clear colonize preference flags
	remove_country_flag = overmind_colonize_habitats
	remove_country_flag = overmind_colonize_gaia
}


# --- EXPAND: Build outposts, claim systems ---
overmind_action_expand = {
	overmind_clear_modifiers = yes
	log = "[OVERMIND] Action: EXPAND for [This.GetName]"
	add_modifier = {
		modifier = overmind_expansion_focus
		days = 180
	}
}


# --- BUILD_FLEET: Queue ship construction ---
# Sub-action variable overmind_subaction controls fleet type:
#   0 = balanced (default)
#   1 = corvette swarm
#   2 = battleship focus
overmind_action_build_fleet = {
	overmind_clear_modifiers = yes
	set_country_flag = overmind_aggressive
	log = "[OVERMIND] Action: BUILD_FLEET for [This.GetName]"
	add_modifier = {
		modifier = overmind_military_focus
		days = 180
	}
	# Check sub-action for fleet type preference
	if = {
		limit = { check_variable = { which = overmind_subaction value = 1 } }
		set_country_flag = overmind_fleet_corvettes
		log = "[OVERMIND] Fleet type: corvette swarm"
	}
	else_if = {
		limit = { check_variable = { which = overmind_subaction value = 2 } }
		set_country_flag = overmind_fleet_battleships
		log = "[OVERMIND] Fleet type: battleship focus"
	}
	else = {
		set_country_flag = overmind_fleet_balanced
		log = "[OVERMIND] Fleet type: balanced"
	}
	set_variable = { which = overmind_subaction value = 0 }
}

# --- FOCUS_TECH: Prioritize research ---
# Also enforces Academic Privilege living standard if available
overmind_action_focus_tech = {
	overmind_clear_modifiers = yes
	log = "[OVERMIND] Action: FOCUS_TECH for [This.GetName]"
	add_modifier = {
		modifier = overmind_research_focus
		days = 180
	}
	# Enforce research-friendly policies
	if = {
		limit = {
			NOR = {
				has_ethic = ethic_gestalt_consciousness
				has_policy_flag = academic_privilege
			}
			can_set_policy = {
				policy = living_standard
				option = living_standard_academic_privilege
			}
		}
		set_policy = {
			policy = living_standard
			option = living_standard_academic_privilege
			cooldown = yes
		}
		log = "[OVERMIND] Set Academic Privilege living standard"
	}
}

# --- PREPARE_WAR: Shift to war footing ---
# Also enforces militarist-friendly policies
overmind_action_prepare_war = {
	overmind_clear_modifiers = yes
	set_country_flag = overmind_aggressive
	log = "[OVERMIND] Action: PREPARE_WAR for [This.GetName]"
	add_modifier = {
		modifier = overmind_war_preparation
		days = 360
	}
	# If already at war, escalate to full assault (max combat_bravery)
	if = {
		limit = { is_at_war = yes }
		remove_country_flag = overmind_aggressive
		set_country_flag = overmind_full_assault
		log = "[OVERMIND] At war — full assault mode (combat_bravery=3.0)"
	}
	# Set aggressive war economy policies
	if = {
		limit = {
			can_set_policy = {
				policy = war_philosophy
				option = war_philosophy_unrestricted
			}
		}
		set_policy = {
			policy = war_philosophy
			option = war_philosophy_unrestricted
			cooldown = yes
		}
		log = "[OVERMIND] Set Unrestricted Wars policy"
	}
	if = {
		limit = {
			can_set_policy = {
				policy = economic_policy
				option = economic_policy_militarist
			}
		}
		set_policy = {
			policy = economic_policy
			option = economic_policy_militarist
			cooldown = yes
		}
		log = "[OVERMIND] Set Militarist economic policy"
	}
}
