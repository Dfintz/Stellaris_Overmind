# ============================================================
# Stellaris Overmind — AI Personality Override
#
# This personality is assigned to empires controlled by the
# Overmind.  It dynamically adjusts based on the LLM's current
# directive via country flags set by the action executor.
#
# The personality uses Stellaris' native AI weight/modifier
# system so the built-in AI actually follows the LLM's strategy:
#   - weapon_preferences → our 4.3 meta (kinetic = autocannon)
#   - armor/shields/hull ratios → balanced 4.3 meta
#   - aggressiveness → scaled by war willingness
#   - military_spending → scaled by directive
#   - colony_spending → scaled by directive
#
# The allow block restricts this personality to empires with
# the overmind_active flag, so it never applies to vanilla AI.
# ============================================================

overmind_controlled = {
	# ---- Personality archetype ----
	# hegemon: balanced focus on superiority — fits our strategic AI
	type = hegemon

	# ---- Numeric modifiers ----
	# Base values — conservative defaults, modified by country flags
	aggressiveness = 1.0
	trade_willingness = 0.8
	bravery = 1.0
	combat_bravery = 1.5

	military_spending = 1.0
	colony_spending = 1.0

	# Diplomatic acceptance — neutral base
	federation_acceptance = 0
	nap_acceptance = 0
	defensive_pact_acceptance = 10
	migration_pact_acceptance = 0
	commercial_pact_acceptance = 10
	research_agreement_acceptance = 20

	# Threat perception — alert but not paranoid
	threat_modifier = 1.0
	threat_others_modifier = 0.75
	friction_modifier = 1.0
	claims_modifier = 1.5

	# ---- Ship design meta (4.3.4) ----
	# Autocannon + Plasma is the swarm meta for corvettes/destroyers
	# Kinetic weapon preference ensures autocannon selection
	weapon_preferences = weapon_type_kinetic

	# Balanced defensive slots (4.3 meta: shields slightly favored)
	armor_ratio = 0.3
	shields_ratio = 0.5
	hull_ratio = 0.2

	# ---- Behaviours ----
	behaviour = {
		conqueror = yes
		subjugator = yes
		liberator = no
		opportunist = yes
		slaver = no
		caste_system = no
		uplifter = no
		purger = no
		displacer = no
		infiltrator = no
		dominator = no
		robot_exploiter = yes
		robot_liberator = no
		propagator = no
		multispecies = yes
		crisis_fighter = yes
		crisis_leader = no
	}

	# ---- Allow block ----
	# Only applies to empires with the overmind_active flag
	allow = {
		has_country_flag = overmind_active
		is_country_type = default
	}

	# ---- Weight ----
	# Very high weight ensures this personality is always chosen
	# when the allow conditions are met
	weight_modifier = {
		weight = 10000
		modifier = {
			factor = 0
			NOT = { has_country_flag = overmind_active }
		}
	}
}
